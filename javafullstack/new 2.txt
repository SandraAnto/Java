def fibonacci(n):
    if n <= 0:
        return 0
    elif n == 1:
        return 1
    else:
        return fibonacci(n - 1) + fibonacci(n - 2)

# Calculate population growth for 12 generations
generations = 12
for i in range(generations):
    population = fibonacci(i)
    print(f"Generation {i}: {population} pairs of rabbits")
